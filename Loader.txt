local function Screen()
	local blur = Instance.new("BlurEffect")
	blur.Size = 0
	blur.Parent = Lighting

	local gui = Instance.new("ScreenGui")
	gui.Name = "ArvotheonTitle"
	gui.IgnoreGuiInset = true
	gui.ResetOnSpawn = false
	gui.DisplayOrder = 999999
	gui.Parent = gethui()  -- <- Use gethui() for hidden GUI

	local darkFrame = Instance.new("Frame")
	darkFrame.BackgroundColor3 = Color3.new(0, 0, 0)
	darkFrame.BackgroundTransparency = 1
	darkFrame.Size = UDim2.new(1, 0, 1, 0)
	darkFrame.ZIndex = 0
	darkFrame.Parent = gui

	local textLabel = Instance.new("TextLabel")
	textLabel.Size = UDim2.new(0.4, 0, 0.15, 0)
	textLabel.Position = UDim2.new(0.3, 0, 0.425, 0)
	textLabel.BackgroundTransparency = 1
	textLabel.Text = "Arvotheon"
	textLabel.Font = Enum.Font.GothamBlack
	textLabel.TextScaled = true
	textLabel.TextTransparency = 1
	textLabel.TextColor3 = Color3.fromRGB(255, 255, 255)
	textLabel.ZIndex = 2
	textLabel.Parent = gui

	local gradient = Instance.new("UIGradient")
	gradient.Color = ColorSequence.new{
		ColorSequenceKeypoint.new(0, Color3.fromRGB(0, 174, 255)),
		ColorSequenceKeypoint.new(1, Color3.fromRGB(255, 255, 255))
	}
	gradient.Parent = textLabel

	TweenService:Create(blur, TweenInfo.new(1), {Size = 20}):Play()
	TweenService:Create(darkFrame, TweenInfo.new(1), {BackgroundTransparency = 0.3}):Play()
	task.wait(0.6)
	TweenService:Create(textLabel, TweenInfo.new(1), {TextTransparency = 0}):Play()
	task.wait(3)
	TweenService:Create(textLabel, TweenInfo.new(1), {TextTransparency = 1}):Play()
	TweenService:Create(darkFrame, TweenInfo.new(1), {BackgroundTransparency = 1}):Play()
	TweenService:Create(blur, TweenInfo.new(1), {Size = 0}):Play()
	task.delay(2, function()
		gui:Destroy()
		blur:Destroy()
	end)
end

Screen()
task.wait(3)
